CREATE TABLE coffees_reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,

        REFERENCES coffees(id) 
        ON DELETE CASCADE 
        ON UPDATE CASCADE

CREATE VIEW coffees_with_reviews AS
SELECT
  c.id,
  c.name,
  c.location,
  c.menu_image,
  c.image1,
  c.image2,
  r.name AS reviewer,
  r.order_name,
  r.order_price,
  r.order_review
FROM
  coffees c
  LEFT JOIN coffees_reviews r ON c.id = r.coffee_id;
